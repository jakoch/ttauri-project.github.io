370  void reserve(size_t new_capacity) noexcept
371  {
372  ttlet extra_capacity = static_cast<ssize_t>(new_capacity) - capacity();
373  if (extra_capacity <= 0) {
374  return;
375  }
376 
377  // Add the extra_capacity to the end of the gap.
378  // LLL...RRR
379  // LLL....RRR
380  ttlet new_begin = _allocator.allocate(new_capacity);
381  ttlet new_it_end = new_begin + size();
382  ttlet new_gap_begin = new_begin + left_size();
383  ttlet new_gap_size = new_capacity - size();
384 
385  if (_begin != nullptr) {
386  placement_move(left_begin_ptr(), left_end_ptr(), new_begin);
387  placement_move(right_begin_ptr(), right_end_ptr(), new_gap_begin + new_gap_size);
388  _allocator.deallocate(_begin, capacity());
389  }
390

392  _it_end = new_it_end;
393  _gap_begin = new_gap_begin;
394  _gap_size = new_gap_size;

593  iterator erase(iterator first, iterator last) noexcept
594  {
595  // place the gap after the last iterator, this way we can use the
596  // it_ptr directly because we don't need to skip the gap.
597  tt_axiom(first.buffer() == this);
598  tt_axiom(last.buffer() == this);
599 
600  set_gap_offset(last.it_ptr());
601  ttlet first_p = first.it_ptr();
602  ttlet last_p = last.it_ptr();
603  ttlet erase_size = last_p - first_p;
604 
605  std::destroy(first_p, last_p);
606  _gap_begin = first_p;
607  _gap_size += erase_size;
608  _it_end -= erase_size;
609  return make_gap_buffer_iterator(this, _gap_begin);
610  }

805  void set_gap_offset(value_type *new_gap_begin) noexcept
806  {
807  tt_axiom(is_valid());
808  if (new_gap_begin < _gap_begin) {
809  // Move data left of the original gap to the end of the new gap.
810  // LLL...RRR
811  // LL...LRRR
812  placement_move_within_array(new_gap_begin, _gap_begin, new_gap_begin + _gap_size);
813 
814  } else if (new_gap_begin > _gap_begin) {
815  // Move data right of the original gap to the beginning of the new gap.
816  // LLL...RRR
817  // LLLR...RR
818  placement_move_within_array(_gap_begin + _gap_size, new_gap_begin + _gap_size, _gap_begin);
819  }
820

822  }
