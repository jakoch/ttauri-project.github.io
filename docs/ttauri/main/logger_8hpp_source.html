130 template<log_level Level, basic_fixed_string SourceFile, int SourceLine, basic_fixed_string Fmt, typename... Args>
131 void log(Args &&...args) noexcept
132 {
133  if (!static_cast<bool>(log_level_global.load(std::memory_order::relaxed) & Level)) {
134  return;
135  }
136 
137  // Add messages in the queue, block when full.
138  // * This reduces amount of instructions needed to be executed during logging.
139  // * Simplifies logged_fatal_message logic.
140  // * Will make sure everything gets logged.
141  // * Blocking is bad in a real time thread, so maybe count the number of times it is blocked.
142 
143  // Emplace a message directly on the queue.
144  log_queue.write<"logger_blocked">()->emplace<log_message<Level, SourceFile, SourceLine, Fmt, forward_value_t<Args>...>>(
145  std::forward<Args>(args)...);
146 
147  if (static_cast<bool>(Level & log_level::fatal) || !logger_start()) {
148  // If the logger did not start we will log in degraded mode and log from the current thread.
149  // On fatal error we also want to log from the current thread.
150  [[unlikely]] logger_flush();
151  }
152 
153  if constexpr (static_cast<bool>(Level & log_level::fatal)) {
154  std::terminate();
155 
156  } else if constexpr (static_cast<bool>(Level & log_level::error)) {
157  // Actually logging of tracing will only work when we cleanly unwind the stack and destruct all trace objects.
158  trace_record();
159  }
160 }
