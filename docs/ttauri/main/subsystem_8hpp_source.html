42 tt_no_inline T start_subsystem(std::atomic<T> &check_variable, T off_value, InitFunc init_function, DeinitFunc deinit_function)
43 {
44  ttlet lock = std::scoped_lock(subsystem_mutex);
45 
46  auto old_value = check_variable.load(std::memory_order::acquire);
47  if (old_value != off_value) {
48  // In the short time before the lock the subsystem became available.
49  return old_value;
50  }
51 
52  if (system_status != system_status_type::running) {
53  // Only when the system is running can subsystems be started.
54  // otherwise they have to run in degraded mode.
55  return off_value;
56  }
57 
58  auto new_value = std::forward<InitFunc>(init_function)();
59 
60  if (new_value != off_value) {
61  subsystem_deinit_list.emplace_back(std::forward<DeinitFunc>(deinit_function));
62  }
63 
64  check_variable.store(new_value, std::memory_order::release);
65  return new_value;
66 }
