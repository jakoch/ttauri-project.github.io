27  {
28  minimum_size = std::max(minimum_size, narrow_cast<int>(std::ceil(min_size)));

70  void update(ssize_t index, float minimum_size, float preferred_size, float maximum_size, float margin) noexcept
71  {
72  tt_axiom(index >= 0);
73  tt_axiom(index < std::ssize(items));
74  tt_axiom(index + 1 < std::ssize(margins));
75 
76  items[index].update(minimum_size, preferred_size, maximum_size);
77 
78  ttlet margin_ = narrow_cast<int>(std::ceil(margin));
79  margins[index] = std::max(margins[index], margin_);
80  margins[index + 1] = std::max(margins[index + 1], margin_);

117  void set_size(float total_size) noexcept
118  {
119  ttlet total_size_ = narrow_cast<int>(std::round(total_size));
120  // It is possible that total_size crosses the maximum_size.
121  tt_axiom(total_size_ >= minimum_size());
122 
123  set_items_to_preferred_size();
124 
125  auto grow_by = total_size_ - size();
126  while (grow_by != 0) {

130  if (resize_beyond_maximum) {

132  }

150  void reserve(ssize_t new_size) noexcept
151  {
152  while (std::ssize(items) < new_size) {
153  items.emplace_back();
154  }
155 
156  while (std::ssize(margins) < new_size + 1) {

163 private:
164  /* The margin between the items, margin[0] is the margin
165  * before the first item. margin[items.size()] is the margin
166  * after the last item. margins.size() == items.size() + 1.
167  */
168  std::vector<int> margins;

171  void set_items_to_preferred_size() noexcept
172  {
173  for (auto &&item : items) {
174  item.size = item.preferred_size;
175  }
176  calculate_offset_and_size();

179  [[nodiscard]] int num_items_can_resize(int grow_by) noexcept
180  {
181  if (grow_by > 0) {
182  return narrow_cast<int>(std::ranges::count_if(items, [](ttlet &item) {
183  return item.size < item.maximum_size;
184  }));
185  } else if (grow_by < 0) {
186  return narrow_cast<int>(std::ranges::count_if(items, [](ttlet &item) {
187  return item.size > item.minimum_size;
188  }));
189 
190  } else {
191  return 0;
192  }
193  }

195  [[nodiscard]] void resize_items(int nr_items, int grow_by, bool resize_beyond_maximum) noexcept
196  {
197  tt_axiom(grow_by != 0);
198  tt_axiom(nr_items > 0);
199 
200  auto per_item_grow_by = grow_by / nr_items;
201  if (per_item_grow_by == 0) {
202  per_item_grow_by = grow_by > 0 ? 1 : -1;
203  }
204 
205  for (auto &&item : items) {
206  auto new_item_size = item.size + per_item_grow_by;
207  if (!resize_beyond_maximum) {
208  new_item_size = std::clamp(new_item_size, item.minimum_size, item.maximum_size);
209  }
210 
211  ttlet this_item_grown_by = new_item_size - item.size;
212  item.size = new_item_size;
213 
214  tt_axiom(item.size >= item.minimum_size);
215  if (!resize_beyond_maximum) {
216  tt_axiom(item.size <= item.maximum_size);
217  }
218 
219  if ((grow_by -= this_item_grown_by) == 0) {
220  // All the growth has been spread to the widgets.
221  break;
222  }
223  }
224 
225  calculate_offset_and_size();
226  }
227 
228  void calculate_offset_and_size() noexcept
229  {
230  auto offset = 0;
231  for (ssize_t i = 0; i != std::ssize(items); ++i) {
232  offset += margins[i];
233  items[i].offset = offset;
234  offset += items[i].size;
235  }
236  }
237 
241  [[nodiscard]] int size() const noexcept
242  {
243  if (items.empty()) {
244  return margins.back();
245  } else {
246  tt_axiom(items.back().offset >= 0);
247  return items.back().offset + items.back().size + margins.back();
248  }
249  }
