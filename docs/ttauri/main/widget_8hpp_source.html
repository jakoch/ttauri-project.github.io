242  set_layout_parameters(geo::transformer auto const &local_to_parent, extent2 size, aarectangle const &clipping_rectangle) noexcept
243  {
244  tt_axiom(gui_system_mutex.recurse_lock_count());

246  _local_to_parent = local_to_parent;

249  auto parent_ = reinterpret_cast<widget *>(parent.get());
250  _local_to_window = local_to_parent * parent_->local_to_window();
251  _window_to_local = ~local_to_parent * parent_->window_to_local();

254  _window_to_local = ~local_to_parent;

261  void
262  set_layout_parameters_from_parent(aarectangle child_rectangle, aarectangle parent_clipping_rectangle, float draw_layer_delta) noexcept
263  {
264  tt_axiom(gui_system_mutex.recurse_lock_count());
265  tt_axiom(child_rectangle.extent() >= _minimum_size);
266 
267  ttlet child_translate = translate2{child_rectangle};
268  ttlet child_size = child_rectangle.extent();
269  ttlet rectangle = aarectangle{child_size};
270  ttlet child_clipping_rectangle = intersect(~child_translate * parent_clipping_rectangle, expand(rectangle, margin()));
271 
272  set_layout_parameters(translate_z(draw_layer_delta) * child_translate, child_size, child_clipping_rectangle);
273  }

275  void set_layout_parameters_from_parent(aarectangle child_rectangle) noexcept
276  {
